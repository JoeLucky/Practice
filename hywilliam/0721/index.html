function escapeHtml(str) {
        return str.replace(/[<>”&]/g, function(match) {
            switch (match) {
                case “<”:
                    // HTML会解析出实体，看源码
                    return “<”;
                case “>”:
                    return “>”;
                case “&”:
                    return “&”;
                case “\””: //习惯双引号的话就需要转义，单引号的话就不用了
                    return “"”;
       }
   });
 }
