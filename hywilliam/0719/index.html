Array.prototype.shuffle = function () {
            var len = this.length,
                i,
                temp;
            while(len) {
                i = Math.floor(Math.random()*len--);
                //  随机取得一个位置并与最末尾的值进行互换，标记为已打乱，len-1
                temp = this[len];
                this[len] = this[i];
                this[i] = temp;
            }
            return this;
        };
