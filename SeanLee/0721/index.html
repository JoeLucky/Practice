在 HTML 中，某些字符是预留的。
在 HTML 中不能使用小于号（<）和大于号（>），这是因为浏览器会误认为它们是标签。
如果希望正确地显示预留字符，我们必须在 HTML 源代码中使用字符实体。
最常见的字符实体如空格，实体名称是[& nbsp];
注意：实体名称对大小写敏感

        function escaptHtml(str){
            str = str.replace(/[<>&'"]/g,function(c){//使用正则匹配
                switch (c){
                    case "<" : return "<";//<
                    case ">" : return ">";//>
                    case "&" : return "&";//&
                    case "'" : return "&apos";//IE不支持,'
                    case "\"" : return """;//"
                }
            })
            return str;
        }
        var teststr = '1<3>2&&true';
        console.log(escaptHtml(teststr));// 1<3>2&&true
