1. 以下JS程序的输出是什么

    var a="undefined";//字符串
    var b="false";//字符串
    var c="";//空字符串
    function assert(aVar){
        if(aVar)
            alert(true);
        else
            alert(false);
    }
    assert(a);//true
    assert(b);//true
    assert(c);//false
